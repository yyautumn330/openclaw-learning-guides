- 步长：0.5 小时（30 分钟）
- 显示：00:00 - 12:00 - 24:00

```swift
Slider(
    value: Binding(
        get: { Float(calendar.component(.hour, from: currentTime)) },
        set: { newHour in
            var components = calendar.dateComponents([.year, .month, .day], from: currentTime)
            components.hour = Int(newHour)
            currentTime = calendar.date(from: components)!
        }
    ),
    in: 0...24,
    step: 0.5
)
```
